  s(k = "down", n = 1) {                                                         ; function wrapper for send keystrokes command
    sleep 100
    switch k 
    {
        case "enter"     : send % "{enter}"
        case "u", "up"   : send % "{ up "    n "}"
        case "d", "down" : send % "{ down "  n "}"
        case "l", "left" : send % "{ left "  n "}"
        case "r", "right": send % "{ right " n "}"
        Default          : send % k
    }
    return
  }
